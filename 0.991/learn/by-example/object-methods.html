import ballerina/io;
type Person object {
    public int age;
    public string firstName;
    public string lastName;
    function __init(int age, string firstName, string lastName) {
        self.age = age;
        self.firstName = firstName;
        self.lastName = lastName;
    }
    function getFullName() returns string {
        return self.firstName + " " + self.lastName;
    }
    function checkAndModifyAge(int condition, int a);
};
function Person.checkAndModifyAge(int condition, int a) {
    if (self.age < condition) {
        self.age = a;
    }
}public function main() {
    Person p1 = new(5, "John", "Doe");
    io:println(p1);    io:println(p1.getFullName());    p1.checkAndModifyAge(10, 50);    io:println(p1);
}

Object Methods

Objects can have functions associated with them. These associated functions are known as methods. Methods can be defined inside or outside the object. To define a method outside the object, first declare the method signature inside the object.
