import ballerina/rabbitmq;
rabbitmq:Connection connection = new({ host: "localhost", port: 5672 });listener rabbitmq:ChannelListener channelListener = new(connection);
@rabbitmq:ServiceConfig {
    queueConfig: {
        queueName: "testQueue"
    }
}
service testSimpleConsumer on channelListener {
    resource function onMessage(string message) {
        log:printInfo("The message received: " + message);
    }

import ballerina/log;
import ballerina/rabbitmq;

rabbitmq:Connection connection = new({ host: "localhost", port: 5672 });

Creates a ballerina RabbitMQ connection that allows reusability if necessary.

 
listener rabbitmq:ChannelListener channelListener = new(connection);

@rabbitmq:ServiceConfig {
    queueConfig: {
        queueName: "testQueue"
    }
}

Consumer service listens to the “testQueue” queue.

 
service testSimpleConsumer on channelListener {

Attaches the service to the listener.

 
    resource function onMessage(string message) {
        log:printInfo("The message received: " + message);
    }
}

Gets triggered when a message is received by the queue.

 
# Make sure to have the RabbitMQ broker running.
#
# Navigate to the directory that contains the
# 'rabbitmq-simple-consumer.bal' file and issue the 'ballerina run' command as follows..

$ ballerina run rabbitmq-simple-consumer.bal
