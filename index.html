Why Ballerina

Cloud Native

Network primitives in the language make it simpler to write services and run them in the cloud.

  • More Info

Flexibly Typed

Structural types with support for openness are used both for static typing within a program and for describing service interfaces.

  • More Info

Data Oriented

Type-safe, declarative processing of JSON, XML, and tabular data with language-integrated queries.

  • More Info

Graphical

Programs have both a textual syntax and an equivalent graphical form based on sequence diagrams.

  • More Info

Concurrent

Easy and efficient concurrency with sequence diagrams and language-managed threads without the complexity of asynchronous functions.

  • More Info

Reliable, Maintainable

Explicit error handling, static types, and concurrency safety, combined with a familiar, readable syntax make programs reliable and maintainable.

// Provide high-level abstractions to work with the HTTP protocol 
import ballerina/http;
                                    
// Define an immutable closed record type to describe both the data in memory and data on the wire
// A closed record type allows exclusively what is specified in the definition
// Let's be conservative in what we send
   type Album readonly & record {|
      string id;
      string title;
      string artist;
      decimal price;
   |};
                                    
// A table is a collection of records where each record represents a row of the table
// Each row is uniquly identified by a key
   table key(id) albums = table [
      {id: "1", title: "Blue Train", artist: "John Coltrane", price: 56.99},
      {id: "2", title: "Jeru", artist: "Gerry Mulligan", price: 17.99},
      {id: "3", title: "Sarah Vaughan and Clifford Brown", artist: "Sarah Vaughan", price: 39.99}
   ];
                                    
// HTTP listener receives requests and then routes them to resources 
// A Service declaration specifies the listener, and a collection of remotely accessible methods.
   service / on new http:Listener(8080) {
   // Respond to HTTP GET requests with the path '/albums'
   // Convert the return value to a response payload with the media type 'application/json'
   resource function get albums() returns Album[] {
      return albums.toArray();
   }
                                    
// Respond to HTTP GET requests with the path '/albums/{id}'
// Resource names can be hierarchical with multiple path segments
resource function get albums/[string id]() returns Album | http:NotFound {
   Album? album = albums[id];
   if album is () {
      return {};
   else {
      return album;
        }
   }
                                    
   // Respond to HTTP POST requests with the path '/albums'
   // Request payload is mapped to the application-specific type 'Album'
   resource function post albums(@http:Payload Album album) returns Album {
      // Add the new album to the table.
      albums.add(album);
      return album;
   }
}                            
                                 
                           
                                 
import ballerina/http;
import ballerinax/googleapis.sheets;
                                    
// Declare your configurable variables at module-level
// Supply values at runtime (e.g. by a TOML file)
configurable string githubPAT = ?;                      // '?' means configuration is required
configurable string sheetsAccessToken = ?;
configurable string spreadSheetId = ?;
configurable string sheetName = "Sheet1";               // Configuration is optional
                                    
map httpHeaders = {
   "Accept": "application/vnd.github.v3+json",
   "Authorization": "token " + githubPAT
};
                                    
// Define an open record type to describe the GitHub payload on the wire
type PR record {                                         
   string url;
   string title;
   string state;
   string created_at;
   string updated_at;
   // GitHub PR has fields other than these specified here
   // Open records allow those unspecified fields 
};
                                    
public function main() returns error? {
   // Create a new HTTP client
   http:Client github = check new ("https://api.github.com/repos/");
   // Perform a GET request and convert the JSON payload to the type 'PR'                  
   PR[] prs = check github->get("ballerina-platform/ballerina-lang/pulls", httpHeaders);   
                                    
   // Create a new Spreadsheet client and append a row with column titles
   sheets:Client gsheets = check new ({auth: {token: sheetsAccessToken}});
   check gsheets->appendRowToSheet(spreadSheetId, sheetName,
                  ["Issue", "Title", "State", "Created At", "Updated At"]);
                                    
   // Destructure each record value with a binding pattern
   foreach var {url, title, state, created_at, updated_at} in prs {    
      check gsheets->appendRowToSheet(spreadSheetId, sheetName,
                  [url, title, state, created_at, updated_at]);
   }
}
                                 
                           
                                 
import ballerina/http;
import ballerina/io;
                                    
// Define an open record type to describe both the payload on the wire and the data in memory
type Country record {
   string country;
   int population;
   string continent;
   int cases;
   int deaths;
   // Open records allow fields other than those specified 
};
                                    
// Prints the top 10 countries having the highest case-fatality ratio.
public function main() returns error? {
   http:Client diseaseEp = check new ("https://disease.sh/v3/");
   // Perform a GET request and convert the JSON payload to the type 'Country'      
   Country[] countries = check diseaseEp->get("covid-19/countries");
                                    
   // Process data declaratively with language-integrated queries
   json summary =
         // Destructure each record value with a binding pattern
         from var {country, continent, population, cases, deaths} in countries
            where population >= 100000 && deaths >= 100
            let decimal caseFatalityRatio = deaths / cases * 100
            order by caseFatalityRatio descending
            limit 10
            // Create a JSON object for each filtered country
            select {country, continent, population, caseFatalityRatio};
   io:println(summary);
}
