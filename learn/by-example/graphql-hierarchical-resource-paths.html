import ballerina/graphql;

// This service has multiple resources with hierarchical resource paths.
// There are two resources under the `profile` path. Therefore, a type named
// `profile`, which has two fields (`quote`, and `name`) will be created when the
// GraphQL schema is generated for this service.
service graphql:Service /graphql on new graphql:Listener(9090) {

    // This resource provides a quote for the provided ID.
    resource function get profile/quote() returns string {
        return "I am the one who knocks!";
    }

    // This resource returns the name for the provided ID.
    resource function get profile/name() returns string {
        return "Walter White";
    }
}

Resources with Hierarchical Resource Paths

The resources in Ballerina GraphQL services can have hierarchical resource paths. When a hierarchical resource is found, a new GraphQL type will be created for each intermediate path, which will have its sub paths as fields.

For more information on the underlying package, see the GraphQL package.

This example shows a GraphQL endpoint, which has a profile field of type Person. A GraphQL client can query this service to retrieve specific fields or subfields of the Person object.

import ballerina/graphql;

service graphql:Service /graphql on new graphql:Listener(9090) {

This service has multiple resources with hierarchical resource paths. There are two resources under the profile path. Therefore, a type named profile, which has two fields (quote, and name) will be created when the GraphQL schema is generated for this service.

 
    resource function get profile/quote() returns string {
        return "I am the one who knocks!";
    }

This resource provides a quote for the provided ID.

 
    resource function get profile/name() returns string {
        return "Walter White";
    }
}

This resource returns the name for the provided ID.

 
# To start the service, navigate to the directory that contains the
# `.bal` file, and issue the `bal run` command below.
# Since Ballerina GraphQL runs on a Ballerina HTTP listener, an `http:Listener`

bal run graphql_hierarchical_resource_paths.bal
