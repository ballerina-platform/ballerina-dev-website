import ballerina/io;
import ballerina/test;
@test:Mock {
    moduleName: ".",
    functionName: "intAdd"
}
public function mockIntAdd(int a, int b) returns int {
    io:println("I'm the mock function!");
    return (a - b);
}
@test:Config {}
function testAssertIntEquals() {
    int answer = 0;
    answer = intAdd(5, 3);
    io:println("Function mocking test");
    test:assertEquals(answer, 2, msg = "function mocking failed");
}
public function intAdd(int a, int b) returns int {
    return (a + b);
}

Function Mocks

Mock functions allow you to hide the real function and engage your own functions when running tests. This allows you to isolate your test functions from the rest.
